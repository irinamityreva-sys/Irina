        // Основные переменные игры
        let canvas, ctx;

        // Инициализация игры
        function init() {
            canvas = document.getElementById('mazeCanvas');
            ctx = canvas.getContext('2d');

            rightBtn.addEventListener('click', () => movePlayer(1));
            downBtn.addEventListener('click', () => movePlayer(2));

            // Обработчики клавиатуры

        // Генерация лабиринта
        function generateMaze() {
            // Инициализация лабиринта (все стены)
            maze = Array(mazeHeight).fill().map(() => Array(mazeWidth).fill(1));
            
            // Рекурсивное разделение для генерации лабиринта
            divide(1, 1, mazeWidth-2, mazeHeight-2);
            
            // Установка начальной позиции игрока
            player.x = 1;
            player.y = 1;
            maze[player.y][player.x] = 0;
            
            // Установка выхода в противоположном углу
            exit.x = mazeWidth - 2;
            exit.y = mazeHeight - 2;
            maze[exit.y][exit.x] = 0;
            
            // Создание монстров
            createMonsters();
        }
        
        // Алгоритм рекурсивного разделения для генерации лабиринта
        function divide(x, y, width, height) {
            if (width < 3 || height < 3) return;
            
            // Выбор ориентации (вертикальная или горизонтальная)
            const horizontal = Math.random() > 0.5;
            
            if (horizontal) {
                // Горизонтальная стена
                const wallY = y + Math.floor(Math.random() * (height - 2)) + 1;
                
                // Проход в стене
                const passageX = x + Math.floor(Math.random() * width);
                
                for (let i = x; i < x + width; i++) {
                    if (i !== passageX) {
                        maze[wallY][i] = 1;
                    }
                }
                
                // Рекурсивные вызовы для верхней и нижней частей
                divide(x, y, width, wallY - y);
                divide(x, wallY + 1, width, y + height - wallY - 1);
            } else {
                // Вертикальная стена
                const wallX = x + Math.floor(Math.random() * (width - 2)) + 1;
                
                // Проход в стене
                const passageY = y + Math.floor(Math.random() * height);
                
                for (let i = y; i < y + height; i++) {
                    if (i !== passageY) {
                        maze[i][wallX] = 1;
                    }
                }
                
                // Рекурсивные вызовы для левой и правой частей
                divide(x, y, wallX - x, height);
                divide(wallX + 1, y, x + width - wallX - 1, height);
            }
        }
        
        // Создание монстров
        function createMonsters() {
            monsters = [];
            const monsterCount = Math.min(2 + level, 8); // Увеличиваем количество монстров с уровнем
            
            for (let i = 0; i < monsterCount; i++) {
                let x, y;
                
                // Ищем случайную позицию, не слишком близко к игроку
                do {
                    x = Math.floor(Math.random() * (mazeWidth - 2)) + 1;
                    y = Math.floor(Math.random() * (mazeHeight - 2)) + 1;
                } while (maze[y][x] === 1 || 
                        (Math.abs(x - player.x) < 3 && Math.abs(y - player.y) < 3));
                
                monsters.push({
                    x: x,
                    y: y,
                    speed: 0.05 + (level * 0.01), // Увеличиваем скорость с уровнем
                    moveProgress: 0,
                    direction: Math.floor(Math.random() * 4)
                });
            }
            
            monstersElement.textContent = monsterCount;
        }
        
        // Отрисовка лабиринта
        function drawMaze() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Расчет смещения для центрирования лабиринта
            const offsetX = (canvas.width - mazeWidth * cellSize) / 2;
            const offsetY = (canvas.height - mazeHeight * cellSize) / 2;
            
            // Отрисовка стен
            ctx.fillStyle = '#0f3460';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            ctx.fillStyle = '#1a1a2e';
            for (let y = 0; y < mazeHeight; y++) {
                for (let x = 0; x < mazeWidth; x++) {
                    if (maze[y][x] === 1) {
                        ctx.fillRect(offsetX + x * cellSize, offsetY + y * cellSize, cellSize, cellSize);
                    }
                }
            }
            
            // Отрисовка выхода
            ctx.fillStyle = 'gold';
            ctx.fillRect(offsetX + exit.x * cellSize, offsetY + exit.y * cellSize, cellSize, cellSize);
            ctx.fillStyle = '#000';
            ctx.font = '20px Arial';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText('В', offsetX + exit.x * cellSize + cellSize/2, offsetY + exit.y * cellSize + cellSize/2);
            
            // Отрисовка игрока
            const playerScreenX = offsetX + player.x * cellSize + cellSize/2;
            const playerScreenY = offsetY + player.y * cellSize + cellSize/2;
            
            ctx.fillStyle = '#4ecca3';
            ctx.beginPath();
            ctx.arc(playerScreenX, playerScreenY, cellSize/3, 0, Math.PI * 2);
            ctx.fill();
            
            // Отрисовка направления игрока
            ctx.strokeStyle = '#fff';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(playerScreenX, playerScreenY);
            
            const directionX = Math.sin(playerDirection * Math.PI / 2);
            const directionY = -Math.cos(playerDirection * Math.PI / 2);
            
            ctx.lineTo(
                playerScreenX + directionX * cellSize/2,
                playerScreenY + directionY * cellSize/2
            );
            ctx.stroke();
            
            // Отрисовка монстров
            monsters.forEach(monster => {
                const monsterScreenX = offsetX + monster.x * cellSize + cellSize/2;
                const monsterScreenY = offsetY + monster.y * cellSize + cellSize/2;
                
                ctx.fillStyle = '#e94560';
                ctx.beginPath();
                ctx.arc(monsterScreenX, monsterScreenY, cellSize/2.5, 0, Math.PI * 2);
                ctx.fill();
                
                // Глаза монстра
                ctx.fillStyle = '#000';
                ctx.beginPath();
                ctx.arc(monsterScreenX - cellSize/6, monsterScreenY - cellSize/6, cellSize/8, 0, Math.PI * 2);
                ctx.arc(monsterScreenX + cellSize/6, monsterScreenY - cellSize/6, cellSize/8, 0, Math.PI * 2);
                ctx.fill();
            });
        }
        
        // Движение игрока
        function movePlayer(direction) {
            if (!gameRunning || gamePaused || isMoving) return;
            
            playerDirection = direction;
            moveQueue.push(direction);
        }
        
        // Обработка движения игрока
        function processPlayerMovement() {
            if (moveQueue.length === 0 || isMoving) return;
            
            const direction = moveQueue.shift();
            playerDirection = direction;
            
            let newX = player.x;
            let newY = player.y;
            
            switch (direction) {
                case 0: // Вверх
                    newY--;
                    break;
                case 1: // Вправо
                    newX++;
                    break;
                case 2: // Вниз
                    newY++;
                    break;
                case 3: // Влево
                    newX--;
                    break;
            }
            
            // Проверка, можно ли двигаться в выбранном направлении
            if (newX >= 0 && newX < mazeWidth && newY >= 0 && newY < mazeHeight && maze[newY][newX] === 0) {
                isMoving = true;
                playerMoveProgress = 0;
            }
        }
        
        // Обновление позиции игрока
        function updatePlayerPosition() {
            if (!isMoving) return;
            
            playerMoveProgress += playerSpeed;
            
            if (playerMoveProgress >= 1) {
                // Завершение движения
                switch (playerDirection) {
                    case 0: // Вверх
                        player.y--;
                        break;
                    case 1: // Вправо
                        player.x++;
                        break;
                    case 2: // Вниз
                        player.y++;
                        break;
                    case 3: // Влево
                        player.x--;
                        break;
                }
                
                isMoving = false;
                
                // Проверка достижения выхода
                if (player.x === exit.x && player.y === exit.y) {
                    completeLevel();
                }
                
                // Проверка столкновения с монстрами
                checkMonsterCollision();
            }
        }
        
        // Движение монстров
        function moveMonsters() {
            monsters.forEach(monster => {
                if (monster.moveProgress > 0) {
                    monster.moveProgress += monster.speed;
                    
                    if (monster.moveProgress >= 1) {
                        // Завершение движения монстра
                        switch (monster.direction) {
                            case 0: // Вверх
                                monster.y--;
                                break;
                            case 1: // Вправо
                                monster.x++;
                                break;
                            case 2: // Вниз
                                monster.y++;
                                break;
                            case 3: // Влево
                                monster.x--;
                                break;
                        }
                        
                        monster.moveProgress = 0;
                        
                        // Проверка столкновения с игроком
                        checkMonsterCollision();
                    }
                } else {
                    // Выбор нового направления для монстра
                    const directions = [];
                    
                    // Проверка возможных направлений
                    if (monster.y > 0 && maze[monster.y-1][monster.x] === 0) directions.push(0); // Вверх
                    if (monster.x < mazeWidth-1 && maze[monster.y][monster.x+1] === 0) directions.push(1); // Вправо
                    if (monster.y < mazeHeight-1 && maze[monster.y+1][monster.x] === 0) directions.push(2); // Вниз
                    if (monster.x > 0 && maze[monster.y][monster.x-1] === 0) directions.push(3); // Влево
                    
                    if (directions.length > 0) {
                        // С вероятностью 70% монстр движется к игроку
                        if (Math.random() < 0.7) {
                            // Находим направление, приближающее к игроку
                            let bestDirection = monster.direction;
                            let minDistance = Infinity;
                            
                            directions.forEach(dir => {
                                let testX = monster.x;
                                let testY = monster.y;
                                
                                switch (dir) {
                                    case 0: testY--; break;
                                    case 1: testX++; break;
                                    case 2: testY++; break;
                                    case 3: testX--; break;
                                }
                                
                                const distance = Math.abs(testX - player.x) + Math.abs(testY - player.y);
                                
                                if (distance < minDistance) {
                                    minDistance = distance;
                                    bestDirection = dir;
                                }
                            });
                            
                            monster.direction = bestDirection;
                        } else {
                            // Случайное направление
                            monster.direction = directions[Math.floor(Math.random() * directions.length)];
                        }
                        
                        monster.moveProgress = 0.001; // Начинаем движение
                    }
                }
            });
        }
        
        // Проверка столкновения с монстрами
        function checkMonsterCollision() {
            for (const monster of monsters) {
                if (monster.x === player.x && monster.y === player.y) {
                    gameOver();
                    return;
                }
            }
        }
        
        // Обработка нажатия клавиш
        function handleKeyDown(e) {
            if (!gameRunning || gamePaused) return;
            
            switch (e.key) {
                case 'w':
                case 'ArrowUp':
                    movePlayer(0);
                    break;
                case 'd':
                case 'ArrowRight':
                    movePlayer(1);
                    break;
                case 's':
                case 'ArrowDown':
                    movePlayer(2);
                    break;
                case 'a':
                case 'ArrowLeft':
                    movePlayer(3);
                    break;
            }
        }
        
        // Запуск игры
        function startGame() {
            startScreen.classList.add('hidden');
            gameRunning = true;
            gamePaused = false;
            level = 1;
            gameTime = 0;
            levelElement.textContent = level;
            timeElement.textContent = gameTime;
            
            generateMaze();
            startTimer();
        }
        
        // Завершение уровня
        function completeLevel() {
            if (level >= 10) {
                // Победа в игре
                winGame();
            } else {
                // Переход на следующий уровень
                gameRunning = false;
                clearInterval(timerInterval);
                completedLevel.textContent = level;
                levelCompleteScreen.classList.remove('hidden');
            }
        }
        
        // Следующий уровень
        function nextLevel() {
            levelCompleteScreen.classList.add('hidden');
            gameRunning = true;
            level++;
            levelElement.textContent = level;
            
            // Увеличиваем сложность
            cellSize = Math.max(25, 40 - level); // Уменьшаем размер клетки для большего лабиринта на экране
            
            generateMaze();
            startTimer();
        }
        
        // Конец игры
        function gameOver() {
            gameRunning = false;
            clearInterval(timerInterval);
            failedLevel.textContent = level;
            gameOverScreen.classList.remove('hidden');
        }
        
        // Победа в игре
        function winGame() {
            gameRunning = false;
            clearInterval(timerInterval);
            winScreen.classList.remove('hidden');
        }
        
        // Перезапуск игры
        function restartGame() {
            levelCompleteScreen.classList.add('hidden');
            gameOverScreen.classList.add('hidden');
            winScreen.classList.add('hidden');
            
            startGame();
        }
        
        // Пауза/продолжение игры
        function togglePause() {
            if (!gameRunning) return;
            
            gamePaused = !gamePaused;
            pauseBtn.textContent = gamePaused ? 'Продолжить' : 'Пауза';
            
            if (gamePaused) {
                clearInterval(timerInterval);
            } else {
                startTimer();
            }
        }
        
        // Запуск таймера
        function startTimer() {
            clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                if (!gamePaused) {
                    gameTime++;
                    timeElement.textContent = gameTime;
                }
            }, 1000);
        }
        
        // Игровой цикл
        function gameLoop() {
            if (gameRunning && !gamePaused) {
                processPlayerMovement();
                updatePlayerPosition();
                moveMonsters();
                drawMaze();
            }
            
            requestAnimationFrame(gameLoop);
        }
        
        // Запуск игры при загрузке страницы
        window.addEventListener('load', init);
    </script>
